% One-pager template — morel-research
%
% Fixed structure (do not reorder):
%   1. Title block (title, authors, date)
%   2. Research question
%   3. Methods
%   4. Results
%   5. Forward-looking
%   6. ONE figure carrying the message
%   7. 3-5 takeaways, with:
%        - >=1 tied to the figure
%        - >=1 on key limitations
%        - any remaining bullets for results that didn't make the figure
%
% Compile with:  make            (uses xelatex via latexmk)
% Requires xelatex for fontspec + Morel typography. Falls back to
% Helvetica Neue if Inter / DM Sans are not installed.
%
% Claude scaffolds this file but does NOT write the prose. The human writes
% the prose. Claude may suggest structure, edits, and questions during
% drafting; the human decides what to incorporate.

\documentclass[10pt]{article}

\usepackage{caption}

% ----- Morel brand palette ---------------------------------------------------
% Source: morel-primordia/projects/morel-life/brand-guide/BRAND-GUIDE.md
\definecolor{morelprimary}{HTML}{C1694F}     % Terracotta
\definecolor{morelaccent}{HTML}{2D5016}      % Forest Green
\definecolor{morelparchment}{HTML}{F5E6D3}   % Parchment Cream
\definecolor{moreldark}{HTML}{3B2F2F}        % Dark Earth (body text)
\definecolor{morelpage}{HTML}{FAFAF8}        % Off-White (page bg)
\definecolor{morelborder}{HTML}{E8D5BF}      % Cream Dark (dividers)
\definecolor{morelmuted}{HTML}{6B5E5E}       % Muted text

% Default text color: warm dark earth, not pure black.
\color{moreldark}

% ----- Typography ------------------------------------------------------------
% Try Inter / DM Sans (brand fonts). Fall back to Helvetica Neue if not
% installed, so the template compiles on a vanilla TeX install.
\IfFontExistsTF{Inter}{%
  \setmainfont{Inter}[
    UprightFont = * Regular,
    BoldFont    = * SemiBold,
    ItalicFont  = * Italic,
    Scale       = 1.0,
  ]%
}{%
  \IfFontExistsTF{DM Sans}{%
    \setmainfont{DM Sans}%
  }{%
    \setmainfont{Helvetica Neue}%
  }%
}

\IfFontExistsTF{DM Serif Display}{%
  \newfontfamily\morelserif{DM Serif Display}%
}{%
  \newfontfamily\morelserif{Georgia}%
}

% Tighten section spacing so we fit on one page; color headings forest green.
\titlespacing*{\section}{0pt}{0.6em}{0.2em}
\titleformat{\section}{\normalfont\bfseries\large\color{morelaccent}}{}{0pt}{}

\setlist[itemize]{leftmargin=1.2em,topsep=0.2em,itemsep=0.1em,parsep=0pt}

% Header/footer with brand rule.
\pagestyle{fancy}
\fancyhf{}

\fancyfoot[L]{\footnotesize\color{morelmuted}\textbf{\color{morelprimary}morel}\,\textbullet\,research}
\fancyfoot[R]{\footnotesize\color{morelmuted}\paperRepoShort}
